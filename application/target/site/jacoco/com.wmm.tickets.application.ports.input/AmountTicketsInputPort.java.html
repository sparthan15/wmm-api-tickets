package com.wmm.tickets.application.ports.input;

import com.wmm.domain.vo.MovementType;
import com.wmm.tickets.application.ports.output.AmountTicketOutputPort;
import com.wmm.tickets.application.usecase.AmountTicketsUseCase;
import lombok.RequiredArgsConstructor;

@RequiredArgsConstructor
public class AmountTicketsInputPort implements AmountTicketsUseCase {

    private final AmountTicketOutputPort amountTicketOutputPort;

    @Override
    public Double incomes(String userId) {
        return amountTicketOutputPort.getMovementsAmountByUserId(userId, MovementType.CREDIT);
    }

    @Override
    public Double outcomes(String userId) {
        return amountTicketOutputPort.getMovementsAmountByUserId(userId, MovementType.DEBIT);
    }
}
